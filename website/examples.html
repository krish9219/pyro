Fibonacci

Recursive Fibonacci with pattern matching.

fn fib(n: int) -> int {
    match n {
        0 => 0,
        1 => 1,
        _ => fib(n-1) + fib(n-2)
    }
}

for i in 0..20 {
    print(fib(i))
}

Structs & Traits

Object-oriented design with traits.

trait Shape {
    fn area(self) -> float
    fn name(self) -> str
}

struct Circle { radius: float }
struct Rect { w: float, h: float }

impl Shape for Circle {
    fn area(self) -> float {
        3.14159 * self.radius * self.radius
    }
    fn name(self) -> str { "Circle" }
}

impl Shape for Rect {
    fn area(self) -> float { self.w * self.h }
    fn name(self) -> str { "Rectangle" }
}

WebSocket Chat

Real-time chat with WebSockets.

use websocket
use json

wss = websocket::Server::new()
mut clients = []

wss.on_connect(|ws| {
    clients.push(ws)
    ws.send("Welcome!")
})

wss.on_message(|ws, msg| {
    data = json::parse(msg)
    for c in clients {
        c.send(json::stringify(data))
    }
})

wss.listen(8080)

CSV Analysis

Read CSV, compute statistics, find insights.

use csv
use stats

data = csv::read("sales.csv")
rev = data |> column("revenue")
                |> map(float)

print("Records: " + str(len(rev)))
print("Total:   $" + str(sum(rev)))
print("Mean:    $" + str(stats::mean(rev)))
print("Median:  $" + str(stats::median(rev)))
print("Max:     $" + str(max(rev)))

// Top 5 by revenue
data |> sort_by("revenue", "desc")
     |> take(5)
     |> each(|r| print(r))

Password Hashing

Secure password storage with PBKDF2.

use crypto
use hex

fn hash_password(pw: str) -> (str, str) {
    salt = crypto::random_bytes(16)
    hash = crypto::pbkdf2(pw, salt, 100000)
    (hex::encode(hash), hex::encode(salt))
}

fn verify(pw: str, hash: str, salt: str) -> bool {
    s = hex::decode(salt)
    h = crypto::pbkdf2(pw, s, 100000)
    hex::encode(h) == hash
}

(hash, salt) = hash_password("s3cret")
print(verify("s3cret", hash, salt))   // true
print(verify("wrong", hash, salt))    // false

QuickSort

Classic quicksort implementation.

fn quicksort(arr: List<int>) -> List<int> {
    if len(arr) <= 1 { return arr }

    pivot = arr[0]
    rest = arr[1..]

    lo = rest |> filter(|x| x < pivot)
    hi = rest |> filter(|x| x >= pivot)

    quicksort(lo) + [pivot] + quicksort(hi)
}

data = [38, 27, 43, 3, 9, 82, 10]
print(quicksort(data))
// [3, 9, 10, 27, 38, 43, 82]

Graph BFS

Breadth-first search with the graph module.

use graph
use collections

g = graph::Graph::new()
g.add_edge("A", "B")
g.add_edge("A", "C")
g.add_edge("B", "D")
g.add_edge("C", "D")
g.add_edge("D", "E")

path = graph::bfs(g, "A", "E")
print("Shortest path: ")
path |> join(" -> ") |> print

JWT Authentication

Create and verify JSON Web Tokens.

use jwt
use time

secret = "my-secret-key"

// Create a token
token = jwt::sign({
    "sub": "user_123",
    "role": "admin",
    "exp": time::now() + 3600
}, secret)

print("Token: " + token)

// Verify and decode
match jwt::verify(token, secret) {
    Ok(claims) => print("User: " + claims["sub"]),
    Err(e) => print("Invalid: " + e.message())
}
