Pyro Documentation

Welcome to the official documentation for the Pyro programming language. Pyro is a compiled language with 23 keywords, 76 built-in modules, and native C++ speed.

Installation

Install Pyro on your system in under a minute.

Linux

curl -fsSL https://aravindlabs.tech/pyro-lang/install.sh | bash

macOS

brew install aravindlabs/tap/pyro

Windows

winget install AravindLabs.Pyro

Build from Source

git clone https://github.com/krish9219/pyro.git
cd pyro
make build
sudo make install

Verify the installation:

pyro --version
# Pyro 1.0.0 (built from 7,045 lines of C++)

Hello World

Create a file called hello.ro:

// No main() needed — just write code
print("Hello, World!")

Run it:

pyro run hello.ro

Or compile and run the binary:

pyro build hello.ro -o hello
./hello

Your First Program

Let's build a small program that reads a name and greets the user:

use io

print("What is your name?")
name = io::readline()
print("Hello, " + name + "! Welcome to Pyro.")

// Pipe operator example
[1, 2, 3, 4, 5]
    |> map(|x| x * x)
    |> filter(|x| x > 5)
    |> each(|x| print("Square: " + str(x)))

Variables & Types

Pyro is statically typed with type inference. Just assign values directly — no keyword needed. Use mut for mutable variables. The optional let keyword can enforce immutability explicitly.

// Simple assignment (no keyword needed)
name = "Pyro"            // type inferred as str
age = 1                  // int
pi = 3.14159             // float
active = true            // bool
data = nil               // nil value

// Mutable (use mut when you need to reassign)
mut counter = 0
counter = counter + 1

// Optional: use let for explicit immutability
// let locked = 42  // compiler enforces no reassignment

// Collections
list = [1, 2, 3]                        // List<int>
map = { "a": 1, "b": 2 }               // Map<str, int>
tuple = (1, "hello", true)              // (int, str, bool)

Functions

Functions are first-class citizens in Pyro. Declare them with fn.

// Basic function
fn greet(name: str) -> str {
    return "Hello, " + name + "!"
}

// Implicit return (last expression)
fn add(a: int, b: int) -> int {
    a + b
}

// Default parameters
fn power(base: int, exp: int = 2) -> int {
    mut result = 1
    for _ in 0..exp {
        result = result * base
    }
    result
}

// Higher-order functions
fn apply_twice(f: fn(int) -> int, val: int) -> int {
    f(f(val))
}

print(greet("World"))
    print(power(3))         // 9 (default exp=2)
    print(power(2, 10))    // 1024
    print(apply_twice(|x| x * 2, 3))  // 12
}

Control Flow

// If / else (expressions — they return values)
status = if score >= 90 { "A" }
             else if score >= 80 { "B" }
             else { "C" }

// For loops
for i in 0..10 {
    print(i)
}

for item in my_list {
    print(item)
}

// While loop
mut n = 10
while n > 0 {
    print(n)
    n = n - 1
}

// Infinite loop with break
loop {
    input = io::readline()
    if input == "quit" { break }
    print("You said: " + input)
}

Structs

struct Point {
    x: float,
    y: float
}

impl Point {
    fn new(x: float, y: float) -> Point {
        Point { x, y }
    }

    fn distance(self, other: Point) -> float {
        dx = self.x - other.x
        dy = self.y - other.y
        math::sqrt(dx * dx + dy * dy)
    }

    fn to_string(self) -> str {
        "(" + str(self.x) + ", " + str(self.y) + ")"
    }
}

a = Point::new(0.0, 0.0)
    b = Point::new(3.0, 4.0)
    print(a.distance(b))   // 5.0
}

Pattern Matching

fn classify(n: int) -> str {
    match n {
        0          => "zero",
        1..10      => "small",
        10..100    => "medium",
        100..1000  => "large",
        _          => "huge"
    }
}

// Destructuring in match
fn process(point: (int, int)) {
    match point {
        (0, 0)    => print("origin"),
        (x, 0)    => print("on x-axis at " + str(x)),
        (0, y)    => print("on y-axis at " + str(y)),
        (x, y)    => print("at (" + str(x) + ", " + str(y) + ")")
    }
}

Traits

trait Printable {
    fn to_string(self) -> str
}

trait Area {
    fn area(self) -> float
}

struct Circle { radius: float }

impl Area for Circle {
    fn area(self) -> float {
        math::PI * self.radius * self.radius
    }
}

impl Printable for Circle {
    fn to_string(self) -> str {
        "Circle(r=" + str(self.radius) + ")"
    }
}

Pipe Operator

The pipe operator |> passes the result of the left expression as the first argument to the right function. This makes data transformations read naturally from left to right.

// Without pipes (nested, hard to read)
print(join(sort(filter(map(data, square), is_even)), ", "))

// With pipes (clear, left to right)
data |> map(square)
     |> filter(is_even)
     |> sort
     |> join(", ")
     |> print

Closures

// Closures capture their environment
fn make_adder(n: int) -> fn(int) -> int {
    |x| x + n  // captures 'n'
}

add5 = make_adder(5)
    print(add5(10))  // 15

    // Multi-line closure
    process = |items| {
        items |> filter(|x| x > 0)
              |> map(|x| x * 2)
              |> sort
    }
}

Error Handling

use fs
use errors

fn read_config(path: str) -> Result<str, Error> {
    content = fs::read(path)?   // ? propagates errors
    if content.is_empty() {
        return Err(errors::new("Config file is empty"))
    }
    Ok(content)
}

match read_config("config.toml") {
        Ok(data)  => print("Config loaded: " + data),
        Err(err) => print("Error: " + err.message())
    }
}

Async / Await

use async
use http

async fn fetch(url: str) -> str {
    resp = await http::get(url)
    resp.body()
}

// Concurrent execution
    (a, b) = await async::join(
        fetch("https://api.example.com/a"),
        fetch("https://api.example.com/b")
    )

    print(a)
    print(b)
}

Type System

Pyro uses a static type system with full type inference. The compiler infers types at compile time, but you can always annotate explicitly for clarity.

// Type inference
x = 42                    // int
y = 3.14                  // float
z = "hello"               // str
list = [1, 2, 3]          // List<int>

// Explicit annotation
a: int = 42
b: List<str> = ["a", "b"]

// Generic types
fn first<T>(list: List<T>) -> Option<T> {
    if list.is_empty() { None }
    else { Some(list[0]) }
}

Grammar Summary

Pyro's grammar is intentionally small. The complete formal grammar fits on a single page:

program     = declaration*
declaration = fn_decl | struct_decl | enum_decl | trait_decl | impl_decl | use_decl | statement
fn_decl     = ["pub"] ["async"] "fn" IDENT "(" params? ")" ["->" type] block
struct_decl = ["pub"] "struct" IDENT "{" fields "}"
enum_decl   = ["pub"] "enum" IDENT "{" variants "}"
trait_decl  = ["pub"] "trait" IDENT "{" fn_sig* "}"
impl_decl   = "impl" [IDENT "for"] IDENT "{" fn_decl* "}"
use_decl    = "use" path ["as" IDENT]
statement   = let_stmt | expr_stmt | return_stmt | for_stmt | while_stmt | loop_stmt
expression  = assignment | pipe | logical | comparison | arithmetic | unary | call | primary
pipe        = expression "|>" expression
